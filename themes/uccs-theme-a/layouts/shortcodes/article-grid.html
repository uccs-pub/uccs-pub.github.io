Use skip="1" to exclude the featured hero article from the grid. limit int Maximum number of cards to show (default 12). heading string Optional section label above the grid. Reads site.RegularPages, excludes standalone pages, sorts newest-first. Delegates card rendering to article-card partial. */ -}} {{- $skip := .Get "skip" | default 0 | int -}} {{- $limit := .Get "limit" | default 12 | int -}} {{- $heading := .Get "heading" -}} {{- $pages := where site.RegularPages "Params.standalone" "ne" true -}} {{- $pages = $pages.ByDate.Reverse -}} {{- $pages = after $skip $pages -}} {{- $pages = first $limit $pages -}} {{ if $heading }}
